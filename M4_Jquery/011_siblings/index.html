các phương thức tìm kiếm anh chị em trong jquery
    siblings() => sẽ trả về tất cả các thẻ anh chị em của selector
    next() => sẽ tìm các thẻ anh chị em tiếp theo liền kề của selector
    nextAll() => sẽ chọn tất cả các thẻ anh chị em tiếp theo
    nextUntil() => sẽ chọn tất cả các thẻ anh chị em nằm giữa 2 selector
    prev() => sẽ chọn thẻ anh chị em liền kề phía trước
    prevAll() =>sẽ chọ tất cả các thẻ anh chị em ở phía trước
    prevUntil() => sẽ chọn các thẻ anh chị em nằm giữa 2 selector
